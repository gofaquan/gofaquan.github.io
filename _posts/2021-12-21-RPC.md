## 简单的示例

### RPC的server端

```go	
package main

import (
	"log"
	"net"
	"net/rpc"
)

// HelloService 构造一个HelloService类型方法，其中Hello方法用于实现打印功能
type HelloService struct{}

func (hp *HelloService) Hello(request string, reply *string) error {
	*reply = "hello" + request
	return nil
}

func main() {
	//把HelloService对象注册为一个服务
	rpc.RegisterName("HelloService", new(HelloService))
	//建立TCP连接
	listener, err := net.Listen("tcp", ":1234")
	if err != nil {
		log.Fatalln("ListenTCP", err)
	}
	conn, err := listener.Accept()
	if err != nil {
		log.Fatalln("ListenTCP", err)
	}

	//运行Server
	rpc.ServeConn(conn)
}

```



### RPC的client端

````go
package main

import (
	"fmt"
	"log"
	"net/rpc"
)

func main() {
	//首先是rpc.Dial拨号服务
	client, err := rpc.Dial("tcp", ":1234")
	if err != nil {
		log.Fatalln("dialing:", err)
	}

	//通过client调用具体的RPC方法
	//第一个参数是用 . 连接的rpc服务名字和方法名
	//第二个和第三个参数是我们自己在在方法里面定义的传入参数
	var reply string
	err = client.Call("HelloService.Hello", "gofaquan", &reply)
	if err != nil {
		log.Fatalln(err)
	}
	fmt.Println(reply)
}

````

输出为：hello：gofaquan 



在实际中，生产肯定更加复杂，接下来我们再来模拟一个 `JSON` RPC调用

### JSON server

```go
package main

import (
	"fmt"
	"log"
	"net"
	"net/rpc"
	"net/rpc/jsonrpc"
	"os"
)

// Arith 算数运算结构体
type Arith struct {
}

// ArithRequest 算数运算请求结构体
type ArithRequest struct {
	A int
	B int
}

// ArithResponse 算数运算响应结构体
type ArithResponse struct {
	Pro int // 乘积
	Quo int // 商
	Rem int // 余数
}

// 乘法运算方法
func (*Arith) Multiply(req ArithRequest, res *ArithResponse) error {
	res.Pro = req.A * req.B
	return nil
}

func main() {
	rpc.Register(new(Arith)) // 注册rpc服务

	lis, err := net.Listen("tcp", "127.0.0.1:8080")
	if err != nil {
		log.Fatalln("fatal error: ", err)
	}

	fmt.Fprintf(os.Stdout, "%s", "start connection")

	conn, err := lis.Accept() // 接收客户端连接请求
	if err != nil {
		log.Fatalln(err)
	}

	// 并发处理客户端请求
	fmt.Fprintf(os.Stdout, "%s", "new client in coming\n")
	jsonrpc.ServeConn(conn)

}

```



### JSON client

```go
package main

import (
	"fmt"
	"log"
	"net/rpc/jsonrpc"
)

func main() {
	conn, err := jsonrpc.Dial("tcp", "127.0.0.1:8080")
	if err != nil {
		log.Fatalln("dailing error: ", err)
	}

	// 算数运算请求结构体
	type ArithRequest struct {
		A int
		B int
	}

	// 算数运算响应结构体
	type ArithResponse struct {
		Pro int // 乘积
		Quo int // 商
		Rem int // 余数
	}
	//这里要传入与原方法一样的形式参数，所以要传入这个结构体的参数
	req := ArithRequest{9, 2}
	var res ArithResponse

	err = conn.Call("Arith.Multiply", req, &res) // 乘法运算
	if err != nil {
		log.Fatalln("arith error: ", err)
	}
	fmt.Printf("%d * %d = %d\n", req.A, req.B, res.Pro)
}

```
